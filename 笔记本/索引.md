### 1️⃣ 数据清洗类

#### 评论数据通用清洗脚本.py

**用途**：清洗评论数据，过滤无效内容

**输入**：原始评论数据 DataFrame

**输出**：清洗后的 DataFrame

**核心函数**：
- `df.columns.str.strip()` - 列名去空格
- `df.drop_duplicates()` - 去重
- `df['列'].isin(list)` - 匹配特定值
- `~df['列']` - 取反（排除）
- `df['列'].str.len()` - 字符串长度
- `df['列'].str.lower()` - 转小写

**关键逻辑**：
```python
# 1. 列名去空格
df.columns = df.columns.str.strip()

# 2. 定义要排除的无效内容
exclude_list = ['该用户没有填写评价', '该用户没有填写内容', '系统默认好评']

# 3. 排除无效内容
df = df[~df['评价内容'].isin(exclude_list)]

# 4. 过滤空值
df = df[df['评价内容'].str.len() > 0]
df = df[df['评价内容'].str.lower() != 'nan']
```

**下次可以这样用**：
```python
def clean_comments(df, target_col, exclude_words=[]):
    """通用评论清洗函数"""
    df = df.copy()
    df.columns = df.columns.str.strip()
    df = df.drop_duplicates()
    
    if target_col in df.columns:
        df[target_col] = df[target_col].astype(str).str.strip()
        if exclude_words:
            df = df[~df[target_col].isin(exclude_words)]
        df = df[df[target_col].str.len() > 0]
    
    return df
```

### 2️⃣ 数据分析类

#### 计算情感得分.py

**用途**：计算评论的情感分数（结合文本和星级）

**输入**：包含评论内容和星级的 DataFrame

**输出**：添加'情感得分'列的 DataFrame

**核心函数**：
- `SnowNLP(text).sentiments` - 中文情感分析
- `df.apply(func, axis=1)` - 逐行应用函数
- `pd.to_numeric()` - 转数值类型

**关键逻辑**：
```python
from snownlp import SnowNLP

def get_sentiment(row):
    content = str(row.get('评价内容', ''))
    star_str = str(row.get('星级', '5 星'))
    
    # 1. 星级转分数
    star_map = {'5 星': 1.0, '4 星': 0.8, '3 星': 0.6, '2 星': 0.4, '1 星': 0.2}
    star_score = star_map.get(star_str, 0.5)
    
    # 2. 文本情感分析
    if len(content) > 5:
        try:
            text_score = SnowNLP(content).sentiments
            # 3. 加权组合（文本 60% + 星级 40%）
            return (text_score * 0.6) + (star_score * 0.4)
        except:
            return star_score
    return star_score

df['情感得分'] = df.apply(get_sentiment, axis=1)
```

**下次可以这样用**：
```python
def add_sentiment_score(df, text_col='内容', weight=0.6):
    """添加情感得分列"""
    from snownlp import SnowNLP
    
    def calc_score(row):
        text = str(row.get(text_col, ''))
        if len(text) > 5:
            try:
                return SnowNLP(text).sentiments * weight + (1-weight) * 0.5
            except:
                return 0.5
        return 0.5
    
    df['情感得分'] = df.apply(calc_score, axis=1)
    return df
```

```python
def analyze_word_frequency(text_series, custom_words=[], stop_words=set()):
    """通用词频分析"""
    import jieba
    from collections import Counter
    
    for word in custom_words:
        jieba.add_word(word)
    
    all_text = "".join(text_series.astype(str).tolist())
    words = jieba.lcut(all_text)
    counts = Counter([w for w in words if len(w) > 1 and w not in stop_words])
    
    return pd.DataFrame(counts.items(), columns=['词语', '频次'])
```

```python
# 根据硬件配置判断是否为 AI PC
def check_ai_hardware(row):
    cpu = str(row.get('CPU 型号', ''))
    gpu = str(row.get('显卡型号', ''))
    
    ai_keywords = ['Ultra', 'NPU', 'AI', 'RTX 40']
    
    if any(k in cpu for k in ai_keywords) or any(k in gpu for k in ai_keywords):
        return 'AI PC'
    return '普通 PC'

df['AI 硬件标签'] = df.apply(check_ai_hardware, axis=1)
```

```python
def add_feature_tags(df, config_cols, keywords_dict):
    """根据配置添加特征标签"""
    def check_tags(row):
        tags = []
        for col in config_cols:
            val = str(row.get(col, ''))
            for tag, keywords in keywords_dict.items():
                if any(k in val for k in keywords):
                    tags.append(tag)
        return ', '.join(set(tags))
    
    df['特征标签'] = df.apply(check_tags, axis=1)
    return df
```

### 4️⃣ ETL 流程类

#### 可分析行提取 ETL.py

**用途**：批量遍历文件夹，提取可分析的数据行

**输入**：文件夹路径（包含多个 Excel 文件）

**输出**：合并后的 DataFrame

**核心函数**：
- `os.walk(folder_path)` - 遍历文件夹树 ⭐
- `pd.read_excel()` - 读取 Excel
- `pd.concat()` - 合并 DataFrame

**关键逻辑**：
```python
import os
import pandas as pd

all_data = []

# 两个 for 循环模式 ⭐
for root, dirs, files in os.walk(folder_path):
    # root = 当前文件夹路径
    # dirs = 子文件夹列表
    # files = 文件列表
    
    for file in files:
        if file.endswith('.xlsx'):
            file_path = os.path.join(root, file)
            
            try:
                df = pd.read_excel(file_path)
                
                # 添加来源标记
                df['来源文件'] = file
                df['来源文件夹'] = os.path.basename(root)
                
                all_data.append(df)
                
            except Exception as e:
                print(f"读取失败 {file}: {e}")

# 合并所有数据
result = pd.concat(all_data, ignore_index=True)
```

**下次可以这样用**：
```python
def batch_read_excel(folder_path, file_pattern='.xlsx'):
    """批量读取文件夹里的所有 Excel"""
    import os
    import pandas as pd
    
    all_data = []
    
    for root, dirs, files in os.walk(folder_path):
        for file in files:
            if file.endswith(file_pattern): # endswith（）
                file_path = os.path.join(root, file)
                try:
                    df = pd.read_excel(file_path)
                    df['_source'] = file
                    all_data.append(df)
                except Exception as e:
                    print(f"Error {file}: {e}")
    
    return pd.concat(all_data, ignore_index=True) if all_data else pd.DataFrame()
```
